// Слушаем Firebase db.collection("lobbies").doc(LOBBY).onSnapshot(doc => { const d = doc.data(); if(d && d.btn) { console.log("Пытаюсь пробить эмулятор:", d.btn); // НАХОДИМ iframe эмулятора const iframe = document.querySelector('iframe') || document.querySelector('iframe'); // Если iframe есть, шлем событие ПРЯМО В НЕГО if (iframe && iframe.contentWindow) { const ev = new KeyboardEvent('keydown', { code: d.btn, bubbles: true, cancelable: true }); iframe.contentWindow.dispatchEvent(ev); // Отпускаем кнопку setTimeout(() => { const up = new KeyboardEvent('keyup', { code: d.btn, bubbles: true }); iframe.contentWindow.dispatchEvent(up); }, 100); } else { // Если iframe нет, шлем в документ (на случай если эмулятор в shadow dom) window.dispatchEvent(new KeyboardEvent('keydown', { code: d.btn, bubbles: true })); } } });
